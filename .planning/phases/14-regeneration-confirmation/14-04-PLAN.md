    ```rust
    use ratatui::Terminal;

    fn test_dialog_snapshot_leave_selected() {
        let backend = TestBackend::new(80, 24);
        let mut terminal = Terminal::new(backend).unwrap();

        let state = ConfirmRegenerateState {
            artifact_index: 0,
            artifact_name: "ssh-key".to_string(),
            affected_targets: vec!["server-1".to_string(), "server-2".to_string()],
            leave_selected: true,
        };

        terminal.draw(|f| {
            regenerate_dialog::render_confirm_regenerate(f, &state, f.area());
        }).unwrap();

        let output = buffer_to_string(terminal.backend().buffer());
        assert_snapshot!(output);
    }
    ```
