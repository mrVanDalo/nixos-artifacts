    ```rust
    Msg::GeneratorFinished { artifact_index, result: Err(error) } => {
        // Set failed status
    }
    ```
    
    Enhance to capture full context:
    ```rust
    Msg::GeneratorFinished { artifact_index, result: Err(error) } => {
        let mut new_model = model.clone();
        if let Some(entry) = new_model.entries.get_mut(artifact_index) {
            entry.status = ArtifactStatus::Failed {
                error: error.clone(),
                output: entry.step_logs.to_string(), // Include accumulated logs
                retry_available: true,

            entry.step_logs.error(format!("Generator failed: {}", error));
        }
        (new_model, Effect::None)
    }
    ```
