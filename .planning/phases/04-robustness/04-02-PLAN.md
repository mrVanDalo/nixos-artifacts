        backend: BackendConfiguration,
        make: BackendConfiguration,
        mut model: Model,
    ) -> Result<RunResult> {

        let child_token = shutdown_token.child_token();

                shutdown_for_signal.cancel();
            }
        });

  <name>Task 2: Implement graceful shutdown on quit</name>
  <files>pkgs/artifacts/src/tui/runtime.rs</files>
  <action>
    Replace the simple `break` on quit with a full shutdown sequence.
    
    Current pattern:
    ```rust
    if effect.is_quit() {
        break;
    }
    ```
    
    New graceful shutdown:
    ```rust
    if effect.is_quit() || shutdown_token.is_cancelled() {
        log_component("RUNTIME", "Initiating graceful shutdown");
        
        // 1. Signal background to shut down after current work
        shutdown_token.cancel();
        
        // 2. Drain any pending results (graceful completion)
        let drain_start = std::time::Instant::now();
        let drain_timeout = std::time::Duration::from_secs(5);
        
        loop {
            // Check timeout
            if drain_start.elapsed() > drain_timeout {
                log_component("RUNTIME", "Drain timeout reached, proceeding with exit");
                break;
            }
            
            // Try to receive with short timeout
            match tokio::time::timeout(
                std::time::Duration::from_millis(100),
                res_rx.recv()
            ).await {
                Ok(Some(result)) => {
                    // Process final result
                    let msg = result_to_message(result);
                    let (new_model, _) = update(model, msg);
                    model = new_model;
                    
                    // Re-render to show final state
                    terminal.draw(|f| render(f, &model))?;
                }
                Ok(None) => {
                    // Channel closed, background exited
                    log_component("RUNTIME", "Result channel closed");
                    break;
                }
                Err(_) => {
                    // Timeout, check if we should continue waiting
                    // Background may still be working
                    continue;
                }
            }
        }
        
        // 3. Drop command channel to signal no more commands
        drop(cmd_tx);
        log_component("RUNTIME", "Command channel dropped");
        
        // 4. Break out of main loop - channels will be dropped naturally
        break;
    }
    ```
    
    This ensures:
    - Background receives shutdown signal
    - Pending results are processed (with timeout)
    - Channels are properly closed
    - Handler is dropped (cleaning temp directories)
  </action>

  </verify>
