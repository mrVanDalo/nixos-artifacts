  <name>Task 3: Convert background loop to tokio::select!</name>
  <files>pkgs/artifacts/src/tui/background.rs</files>
  <action>
    Convert the background task's while loop to a tokio::select! loop that listens for both commands and cancellation.
    
    Current loop:
    ```rust
    tokio::spawn(async move {
        let mut handler = BackgroundEffectHandler::new(backend, make);
        while let Some(cmd) = rx_cmd.recv().await {
            let result = handler.execute(cmd).await;
            if tx_res.send(result).is_err() {
                break;
            }
        }
    });
    ```
    
    New select!-based loop:
    ```rust
    use tokio_util::sync::CancellationToken;
    
    tokio::spawn(async move {
        let mut handler = BackgroundEffectHandler::new(backend, make);
        log_component("BACKGROUND", "Task started, listening for commands");
        
        loop {
            tokio::select! {
                // Check for shutdown signal first
                _ = shutdown_token.cancelled() => {
                    log_component("BACKGROUND", "Shutdown requested, finishing current work");
                    // Process any remaining commands in queue before exiting
                    while let Ok(cmd) = rx_cmd.try_recv() {
                        log_component("BACKGROUND", "Processing queued command before shutdown");
                        let result = handler.execute(cmd).await;
                        let _ = tx_res.send(result); // Best effort
                    }
                    log_component("BACKGROUND", "Exiting cleanly");
                    break;
                }
                
                // Process next command
                Some(cmd) = rx_cmd.recv() => {
                    log_component("BACKGROUND", &format!("Executing command: {:?}", cmd));
                    let result = handler.execute(cmd).await;
                    log_component("BACKGROUND", "Command complete, sending result");
                    if tx_res.send(result).is_err() {
                        log_component("BACKGROUND", "TUI closed, exiting");
                        break;
                    }
                }
                
                // Channel closed
                else => {
                    log_component("BACKGROUND", "Command channel closed, exiting");
                    break;
                }
            }
        }
        
        // Handler is dropped here, cleaning up temp directories
        log_component("BACKGROUND", "Task complete, handler dropped");
    });
    ```
    
    This maintains FIFO ordering while adding cooperative cancellation.
  </action>

  </verify>
