    ```rust
    fn create_machines_json_file(
        make: &MakeConfiguration,
        nixos_targets: &[String],
        backend_name: &str,
    ) -> Result<(TempFile, PathBuf)> {
        let dir = TempFile::new_dir("machines")?;
        let path = dir.join("machines.json");
        let config: serde_json::Map<String, serde_json::Value> = nixos_targets
            .iter()
            .map(|machine| {
                let config = make
                    .get_backend_config_for(machine, backend_name)
                    .map(|m| serde_json::to_value(m).unwrap_or(json!({})))
                    .unwrap_or(json!({}));
                (machine.clone(), config)
            })
            .collect();
        let text = to_string_pretty(&config)?;
        fs::write(&path, &text)
            .with_context(|| format!("writing {}", path.display()))?;
        Ok((dir, path))
    }
    ```

    2. create_users_json_file (~20 lines) - similar pattern for home_targets

    3. create_config_json_file (~15 lines) - extracted from run_serialize and run_check_serialization

    Update run_shared_serialize, run_shared_check_serialization, run_serialize, and run_check_serialization to use these helpers.

    Each helper should handle its own error context and return both the TempFile (for lifetime) and the PathBuf.

    ```rust
        script_path: &Path,
        out_dir: &Path,
        config_path: &Path,
        context: &str,
        target: &str,
    ) -> std::process::Command {
        let mut cmd = std::process::Command::new("sh");
        cmd.arg(script_path)
            .env("out", out_dir)
            .env("config", config_path)
            .env("artifact_context", context)
            .stdout(Stdio::piped())
            .stderr(Stdio::piped());

        if context == "homemanager" {
            cmd.env("username", target);
        } else {
            cmd.env("machine", target);
        }
        cmd
    }
    ```
