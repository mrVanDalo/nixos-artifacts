    ```rust
    #[cfg(feature = "logging")]
    #[macro_export]
    macro_rules! debug {
        ($($arg:tt)*) => {
            $crate::logging::Logger::global().log(
                $crate::logging::LogLevel::Debug,
                module_path!(),
                line!(),
                format!($($arg)*)
            )
        };
    }

    #[cfg(not(feature = "logging"))]
    #[macro_export]
    macro_rules! debug {
        ($($arg:tt)*) => {};
    }
    ```

    The format! call inside macros ensures the message is only constructed when the macro is invoked.

    - Log level in brackets

    ```rust
    pub struct Logger {
        file: Mutex<std::fs::File>,
        min_level: LogLevel,
        path: PathBuf,
    }

    impl Logger {
        /// Create new logger from CLI args - validates file writability at startup
        pub fn new_from_args(args: &Cli) -> anyhow::Result<Option<Self>>> {
            // Only enabled when --log-file provided
            let path = match &args.log_file {
                Some(p) => p,
                None => return Ok(None),
            };

            // Fail fast validation
            Self::validate_path(path)?;

            // Open file (overwrite mode, create dirs)
            let file = OpenOptions::new()
                .create(true)
                .write(true)
                .truncate(true)
                .open(path)?;

            // Set permissions to 640
            #[cfg(unix)]
            {
                use std::os::unix::fs::PermissionsExt;
                let mut perms = file.metadata()?.permissions();
                perms.set_mode(0o640);
                file.set_permissions(perms)?;
            }

            Ok(Some(Self {
                file: Mutex::new(file),
                min_level: args.log_level,
                path: path.clone(),
            }))
        }

        /// Validate path is writable (fail fast)
        fn validate_path(path: &Path) -> anyhow::Result<()> {
            // Check if path is a directory
            if path.is_dir() {
                return Err(anyhow!("Log path cannot be a directory: {}", path.display()));
            }

            // Check parent directory exists or can be created
            if let Some(parent) = path.parent() {
                if !parent.exists() {
                    std::fs::create_dir_all(parent)?;
                }
                if !parent.is_dir() {
                    return Err(anyhow!("Log path parent is not a directory: {}", parent.display()));
                }
                // Try to write a test byte
                let test_file = parent.join(".artifacts_log_test");
                match std::fs::File::create(&test_file) {
                    Ok(_) => { let _ = std::fs::remove_file(&test_file); }
                    Err(e) => return Err(anyhow!("Cannot write to log directory: {}", e)),
                }
            }

            Ok(())
        }
    }
    ```

    - Return None when --log-file not provided (logging disabled)
    - Validate file writability immediately (fail fast)
    - Create parent directories if needed

    - Overwrite existing file (not append)

    ```rust
    impl Logger {
        /// Log a message at the specified level
        pub fn log(&self,
            level: LogLevel,
            module_path: &str,
            line: u32,
            message: String
        ) {
            // Filter by level
            if level < self.min_level {
                return;
            }

            // Build log entry
            let timestamp = Self::format_timestamp();
            let level_str = format!("[{:?}]", level).to_uppercase();

            // Format: [TIMESTAMP] [LEVEL] module: message
            // At DEBUG level, include line number
            let entry = if level == LogLevel::Debug {
                format!("[{}] [{}] {}:{}: {}",
                    timestamp, level_str, module_path, line, message)
            } else {
                format!("[{}] [{}] {}: {}",
                    timestamp, level_str, module_path, message)
            };

            // Write with lock, flush immediately
            if let Ok(mut file) = self.file.lock() {
                let _ = writeln!(file, "{}", entry);
                let _ = file.flush();
            }
        }

        /// Format timestamp as ISO8601 with milliseconds: 2024-01-20T14:32:10.123Z
        fn format_timestamp() -> String {
            use std::time::SystemTime;
            let now = SystemTime::now()
                .duration_since(SystemTime::UNIX_EPOCH)
                .unwrap_or_default();
            let secs = now.as_secs();
            let millis = now.subsec_millis();

            // Convert to datetime (simplified, use chrono if available)
            // For now, use a simpler format: HH:MM:SS.mmm
            let hours = (secs / 3600) % 24;
            let mins = (secs / 60) % 60;
            let secs = secs % 60;

            format!("{:02}:{:02}:{:02}.{:03}", hours, mins, secs, millis)
        }

        /// Global logger instance (singleton pattern)
        fn global() -> Option<&'static Logger> {
            // Use OnceLock for lazy initialization
            static LOGGER: OnceLock<Option<Logger>> = OnceLock::new();
            LOGGER.get_or_init(|| None).as_ref()
        }

        /// Initialize global logger from CLI args (called once at startup)
        pub fn init_from_args(args: &Cli) -> anyhow::Result<()> {
            let logger = Self::new_from_args(args)?;
            // Store in global... requires interior mutability
            // Use a static Mutex for the Option<Logger>
            Ok(())
        }
    }
    ```

    - Flush after each log entry
    - ISO8601-like timestamp with milliseconds
    - Module path included (DEBUG level adds line number)
    - Level filtering at runtime (based on --log-level)
