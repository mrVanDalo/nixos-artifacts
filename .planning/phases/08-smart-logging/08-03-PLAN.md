    **Do NOT convert:**
    - TUI output (visual feedback is essential)
    - User-facing error messages
    - Generator/serialization script output (shown in TUI)

    **DO convert/remove:**
    - Debug println! statements that dump internal state
    - Temporary debugging eprintln! statements

    Examples:
    ```rust

    println!(">>> DESC: {}", description);

    // NEW - use macros (only when feature enabled)

    debug!("Prompt description: {}", description);
    #[cfg(feature = "logging")]
    error!("Temp file error: {}", error);
    ```

            let temp_dir = TempDir::new().unwrap();
            let log_path = temp_dir.path().join("test.log");
            
            let logger = Logger::new(&log_path, LogLevel::Debug).unwrap();
            logger.log(LogLevel::Info, "test", 1, "test message".to_string());
            
            assert!(log_path.exists());
            let content = std::fs::read_to_string(&log_path).unwrap();
            assert!(content.contains("test message"));
        }

        #[test]
        #[cfg(feature = "logging")]
        fn test_log_format() {
            let temp_dir = TempDir::new().unwrap();
            let log_path = temp_dir.path().join("test.log");
            
            let logger = Logger::new(&log_path, LogLevel::Debug).unwrap();
            logger.log(LogLevel::Info, "my_module", 42, "hello".to_string());
            
            let content = std::fs::read_to_string(&log_path).unwrap();
            // Format: [TIMESTAMP] [LEVEL] module: message
            assert!(content.contains("[INFO]"));
            assert!(content.contains("my_module:"));
            assert!(content.contains("hello"));
        }

        #[test]
        #[cfg(feature = "logging")]
        fn test_level_filtering() {
            let temp_dir = TempDir::new().unwrap();
            let log_path = temp_dir.path().join("test.log");
            
            let logger = Logger::new(&log_path, LogLevel::Warn).unwrap();
            logger.log(LogLevel::Debug, "test", 1, "debug msg".to_string());
            logger.log(LogLevel::Info, "test", 1, "info msg".to_string());
            logger.log(LogLevel::Warn, "test", 1, "warn msg".to_string());
            
            let content = std::fs::read_to_string(&log_path).unwrap();
            assert!(!content.contains("debug msg"));
            assert!(!content.contains("info msg"));
            assert!(content.contains("warn msg"));
        }

        #[test]
        #[cfg(feature = "logging")]
        fn test_file_permissions() {
            use std::os::unix::fs::PermissionsExt;
            
            let temp_dir = TempDir::new().unwrap();
            let log_path = temp_dir.path().join("test.log");
            
            let _logger = Logger::new(&log_path, LogLevel::Debug).unwrap();
            
            let metadata = log_path.metadata().unwrap();
            let perms = metadata.permissions();
            // Should be 0o640
