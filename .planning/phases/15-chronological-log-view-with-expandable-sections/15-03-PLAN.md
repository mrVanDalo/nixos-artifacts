```rust
// In update() match for Screen::ChronologicalLog
(KeyCode::Char(' '), Screen::ChronologicalLog(ref state)) => {
    // Toggle currently focused section
    if let Some(step) = state.focused_section {
        let mut new_state = state.clone();
        new_state.toggle_section(step);
        model.screen = Screen::ChronologicalLog(new_state);
    }
    Effect::None
}

(KeyCode::Enter, Screen::ChronologicalLog(ref state)) => {
    // Same as Space - toggle focused section
    // ... same implementation ...
}

(KeyCode::Char('+') | KeyCode::Char('='), Screen::ChronologicalLog(_)) => {
    model.screen = Screen::ChronologicalLog(ChronologicalLogState {
        expanded_sections: HashSet::from([LogStep::Check, LogStep::Generate, LogStep::Serialize]),
        ..state.clone()
    });
    Effect::None
}

(KeyCode::Char('-'), Screen::ChronologicalLog(_)) => {
    model.screen = Screen::ChronologicalLog(ChronologicalLogState {
        expanded_sections: HashSet::new(),
        ..state.clone()
    });
    Effect::None
}

(KeyCode::Char('j') | KeyCode::Down, Screen::ChronologicalLog(ref state)) => {
    // Move focus to next section
    let mut new_state = state.clone();
    new_state.focus_next();
    model.screen = Screen::ChronologicalLog(new_state);
    Effect::None
}

(KeyCode::Char('k') | KeyCode::Up, Screen::ChronologicalLog(ref state)) => {
    // Move focus to previous section
    let mut new_state = state.clone();
    new_state.focus_previous();
    model.screen = Screen::ChronologicalLog(new_state);
    Effect::None
}

(KeyCode::PageDown, Screen::ChronologicalLog(ref state)) => {
    let mut new_state = state.clone();
    new_state.scroll_down(10); // Scroll 10 lines
    model.screen = Screen::ChronologicalLog(new_state);
    Effect::None
}

(KeyCode::PageUp, Screen::ChronologicalLog(ref state)) => {
    let mut new_state = state.clone();
    new_state.scroll_up(10);
    model.screen = Screen::ChronologicalLog(new_state);
    Effect::None
}

(KeyCode::Esc | KeyCode::Char('q'), Screen::ChronologicalLog(_)) => {
    model.screen = Screen::ArtifactList;
    Effect::None
}
```
