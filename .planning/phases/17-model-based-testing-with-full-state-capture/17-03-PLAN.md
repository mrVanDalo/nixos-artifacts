2. Create model test helper that runs event sequence and captures states:
   ```rust
   fn run_events(mut model: Model, events: Vec<Msg>) -> Vec<(ModelState, String)> {
       let backend = TestBackend::new(80, 24);
       let mut terminal = Terminal::new(backend).unwrap();
       let mut states = Vec::new();
       
       for msg in events {
           // Capture state BEFORE processing
           let state_before = ModelState::from_model(&model);
           
           // Apply the message (pure update function)
           let (new_model, _effect) = update(model, msg);
           model = new_model;
           
           // Render the view
           terminal.draw(|f| {
               match &model.screen {
                   Screen::ArtifactList => render_artifact_list(f, &model, f.area()),
                   _ => {}
               }
           }).unwrap();
           
           let rendered = terminal.backend().to_string();
           states.push((state_before, rendered));
       }
       
       states
   }
   ```

3. Create first navigation test:
   ```rust
   #[test]
   fn test_navigate_down_updates_selection() {
       let model = make_test_model();
       let events = vec![down()];
       
       let states = run_events(model, events);
       
       // Verify state transition
       assert_eq!(states[0].0.selected_index, 0);  // Before: index 0
       assert_eq!(states[0].1, ...);  // Snapshot of rendered view
   }
   ```
