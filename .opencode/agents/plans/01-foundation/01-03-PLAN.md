## Decisions (LOCKED)

### Runtime Architecture

- Main loop uses tokio::select! for concurrent polling
- Polls both: terminal events AND channel results
- Effects sent to background via tx_cmd.send()
- Results converted to Msg and fed into update()

### State Isolation

- Model stays in main thread only
- Background receives immutable data snapshots
- Results come back as messages (pure Elm architecture)

### Error Handling

- Channel send errors: log and continue
- Result processing errors: convert to error messages
- Terminal errors: propagate up

### Migration Strategy

- Replace execute_effect_loop (synchronous blocking) with async loop
- Add AsyncEventSource trait for async event polling
- Convert Effect to EffectCommand before sending
- Convert EffectResult back to Msg on receive

```rust
use std::future::Future;
use std::pin::Pin;

/// Trait for async event sources
pub trait AsyncEventSource {
    /// Get the next event asynchronously
    fn next_event_async(&mut self,
    ) -> Pin<Box<dyn Future<Output = Option<Msg>> + Send + '_>>;
}

/// Async implementation for TerminalEventSource
impl AsyncEventSource for TerminalEventSource {
    fn next_event_async(
        &mut self,
    ) -> Pin<Box<dyn Future<Output = Option<Msg>> + Send + '_>> {
        Box::pin(async move {
            // Use tokio::time::timeout for non-blocking poll
            match tokio::time::timeout(self.tick_rate, tokio::task::spawn_blocking(|| {
                if event::poll(std::time::Duration::from_millis(10)).ok()? {
                    match event::read().ok()? {
                        Event::Key(key) if key.kind == KeyEventKind::Press => {
                            Some(Msg::Key(KeyEvent::from(key)))
                        }
                        _ => Some(Msg::Tick),
                    }
                } else {
                    Some(Msg::Tick)
                }
            })).await {
                Ok(Ok(msg)) => msg,
                Ok(Err(_)) | Err(_) => Some(Msg::Tick),
            }
        })
    }
}
```

Note: This is a basic async wrapper. In production, consider using
crossterm::event::EventStream for true async event handling.

```rust
use crate::tui::background::spawn_background_task;
use tokio::sync::mpsc::UnboundedSender;

pub async fn run_async<B>(
    terminal: &mut Terminal<B>,
    events: &mut dyn AsyncEventSource,
    backend: crate::config::backend::BackendConfiguration,
    make: crate::config::make::MakeConfiguration,
    mut model: Model,
) -> Result<RunResult>
where
    B: Backend,
{
    let mut frames_rendered = 0;
    
    // Spawn background task
    let (tx_cmd, mut rx_res) = spawn_background_task(backend, make);
    
    // Render initial state
    terminal.draw(|f| render(f, &model))?;
    frames_rendered += 1;
    
    // Process initial effect (check serialization)
    let initial_effect = init(&model);
    send_effects_to_background(&initial_effect, &tx_cmd);
    
    loop {
        terminal.draw(|f| render(f, &model))?;
        frames_rendered += 1;
        
        tokio::select! {
            // Terminal events
            msg = events.next_event_async() => {
                if let Some(msg) = msg {
                    let (new_model, effect) = update(model, msg);
                    model = new_model;
                    
                    if effect.is_quit() {
                        break;
                    }
                    
                    send_effects_to_background(&effect, &tx_cmd);
                }
            }
            // Effect results from background
            Some(result) = rx_res.recv() => {
                let msg = result_to_message(result);
                let (new_model, effect) = update(model, msg);
                model = new_model;
                
                if effect.is_quit() {
                    break;
                }
                
                send_effects_to_background(&effect, &tx_cmd);
            }
        }
    }
    
    // Drop sender to signal background task to exit
    drop(tx_cmd);
    
    Ok(RunResult {
        final_model: model,
        frames_rendered,
    })
}

fn send_effects_to_background(effect: &Effect, tx_cmd: &UnboundedSender<EffectCommand>) {
    match effect {
        Effect::Batch(effects) => {
            for e in effects {
                if let Some(cmd) = effect_to_command(e.clone()) {
                    let _ = tx_cmd.send(cmd);
                }
            }
        }
        other => {
            if let Some(cmd) = effect_to_command(other.clone()) {
                let _ = tx_cmd.send(cmd);
            }
        }
    }
}
```
